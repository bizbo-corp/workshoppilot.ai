```tsx
const session = await db.query.sessions.findFirst({ where: eq(sessions.id, sessionId), with: { workshop: true } });
if (!session) redirect('/');
const { workshop } = session;
const { userId } = await auth();
if (!userId) redirect('/');
let isReadOnly = false;
if (workshop.clerkUserId !== userId) {
  const participant = await resolveClerkParticipant(workshop.id);
  if (!participant) redirect('/');
  isReadOnly = true;
}
const rows = await db.select().from(buildPacks)
  .where(and(eq(buildPacks.workshopId, workshop.id), like(buildPacks.title, 'Journey Flow:%')));
const jsonRow = rows.find((r) => r.formatType === 'json' && r.content);
```

Autosave effect (from journey-map-content.tsx lines 53-80 — copy this shape exactly):
```tsx
useEffect(() => {
  if (!isDirty || isReadOnly) return;
  if (saveTimerRef.current) clearTimeout(saveTimerRef.current);
  saveTimerRef.current = setTimeout(async () => {
    try {
      const state = storeApi.getState();
      if (state.nodes.length === 0) return; // never save empty state
      await fetch('/api/build-pack/save-journey-flow', { method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ workshopId, state }) });
      storeApi.getState().markClean();
    } catch (err) { console.error('Autosave failed:', err); }
  }, 2000);
  return () => { if (saveTimerRef.current) clearTimeout(saveTimerRef.current); };
}, [isDirty, workshopId, isReadOnly, storeApi]);
```
NOTE: `JSON.stringify(state)` on the full store includes action functions — they serialize away harmlessly (functions are dropped by JSON.stringify), exactly as the old mapper does. Keep the same approach for consistency.
