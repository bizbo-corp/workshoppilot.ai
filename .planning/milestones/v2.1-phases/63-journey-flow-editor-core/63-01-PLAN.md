```typescript
export type JourneyFlowUiType =
  | 'dashboard' | 'landing-page' | 'form' | 'table' | 'detail-view'
  | 'wizard' | 'modal' | 'settings' | 'auth' | 'onboarding' | 'search' | 'error';

export type JourneyFlowPriority = 'must-have' | 'should-have' | 'nice-to-have';

export interface JourneyFlowNode {
  id: string;                    // 'jf-node-{uuid}'
  name: string;
  uiType: JourneyFlowUiType;
  purpose: string;               // the "short description" shown on the card
  keyElements: string[];
  addressesPain?: string;
  priority: JourneyFlowPriority;
  position: { x: number; y: number };
}

export interface JourneyFlowEdge {
  id: string;                    // 'jf-edge-{uuid}'
  sourceNodeId: string;
  targetNodeId: string;
  sourceHandle?: string;
  targetHandle?: string;
}

export interface JourneyFlowState {
  nodes: JourneyFlowNode[];
  edges: JourneyFlowEdge[];
  isApproved: boolean;
  isDirty: boolean;
  lastSavedAt?: string;
  _schemaVersion: number;        // 1
}
```

Also export display maps (port labels from journey-node-detail-dialog.tsx):
- `UI_TYPE_LABELS: Record<JourneyFlowUiType, string>` (Dashboard, Landing Page, Form, Table, Detail View, Wizard, Modal, Settings, Auth, Onboarding, Search, Error Page)
- `PRIORITY_LABELS: Record<JourneyFlowPriority, string>` (Must Have / Should Have / Nice to Have)
- `DEFAULT_JOURNEY_FLOW_STATE: JourneyFlowState` constant (`{ nodes: [], edges: [], isApproved: false, isDirty: false, _schemaVersion: 1 }`)

No flowType on edges — Journey Flow has a single edge style (fork support comes free: nothing prevents one source having multiple outgoing edges).

Keep this file dependency-free (no React, no zustand imports) so both server and client code can import it.
